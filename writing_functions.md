Write a function to compute z scores.

``` r
z_scores <- function(x) {
  
  if (!is.numeric(x)) {
    stop("The input x should be numeric.")
  }
  
  if (length(x) < 5) {
    stop("Only compute z scores when the input has 5 or more numbers.")
  }
  
  z = (x - mean(x)) / sd(x)
  
  z
}
```

``` r
mean_and_sd <- function() {
  
  
  if (!is.numeric(x)) {
    stop("The input x should be numeric.")
  }
  
  if (length(x) < 5) {
    stop("Only compute z scores when the input has 5 or more numbers.")
  }
  
  mean_x = mean(x, na.rm = TRUE)
  sd_x = sd(x, na.rm = TRUE)
  
  c(mean_x, sd_x)
  
  tibble(
    mean = mean_x,
    sd = sd_x
  )
  
  
  
}
```
